**Implementation (pagination excerpt):**
```cshtml
@if (Model.TotalPages > 1)
{
    <div class="pagination">
        @{
            int current = Model.CurrentPage;
            int total = Model.TotalPages;
            int showPages = 5;
            int start = Math.Max(1, current - showPages / 2);
            int end = Math.Min(total, start + showPages - 1);
            if (end == total) start = Math.Max(1, end - showPages + 1);
        }

        @if (current > 1)
        {
            <a href="@Url.Action("Liked", new { page = 1, search = Model.SearchQuery, genreIds = Model.SelectedGenreIds, yearFrom = Model.YearFrom, yearTo = Model.YearTo })"
               class="page-link">«</a>
            <a href="@Url.Action("Liked", new { page = current - 1, search = Model.SearchQuery, genreIds = Model.SelectedGenreIds, yearFrom = Model.YearFrom, yearTo = Model.YearTo })"
               class="page-link">‹</a>
        }

        @for (int i = start; i <= end; i++)
        {
            var isActive = i == current ? "active" : "";
            <a href="@Url.Action("Liked", new { page = i, search = Model.SearchQuery, genreIds = Model.SelectedGenreIds, yearFrom = Model.YearFrom, yearTo = Model.YearTo })"
               class="page-link @isActive">@i</a>
        }

        @if (current < total)
        {
            <a href="@Url.Action("Liked", new { page = current + 1, search = Model.SearchQuery, genreIds = Model.SelectedGenreIds, yearFrom = Model.YearFrom, yearTo = Model.YearTo })"
               class="page-link">›</a>
            <a href="@Url.Action("Liked", new { page = total, search = Model.SearchQuery, genreIds = Model.SelectedGenreIds, yearFrom = Model.YearFrom, yearTo = Model.YearTo })"
               class="page-link">»</a>
        }

        <span class="page-info">Страница @current из @total</span>
    </div>
}
```
